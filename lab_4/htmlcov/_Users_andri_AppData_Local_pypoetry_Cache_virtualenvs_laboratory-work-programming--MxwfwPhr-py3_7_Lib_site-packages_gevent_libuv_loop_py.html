247 def _start_callback_timer(self): 

248 # The purpose of the callback timer is to ensure that we run 

249 # callbacks as soon as possible on the next iteration of the event loop. 

250 

251 # In libev, we set a 0 duration timer with a no-op callback. 

252 # This executes immediately *after* the IO poll is done (it 

253 # actually determines the time that the IO poll will block 

254 # for), so having the timer present simply spins the loop, and 

255 # our normal prepare watcher kicks in to run the callbacks. 

256 

257 # In libuv, however, timers are run *first*, before prepare 

258 # callbacks and before polling for IO. So a no-op 0 duration 

259 # timer actually does *nothing*. (Also note that libev queues all 

260 # watchers found during IO poll to run at the end (I think), while libuv 

261 # runs them in uv__io_poll itself.) 

262 

263 # From the loop inside uv_run: 

264 # while True: 

265 # uv__update_time(loop); 

266 # uv__run_timers(loop); 

267 # # we don't use pending watchers. They are how libuv 

268 # # implements the pipe/udp/tcp streams. 

269 # ran_pending = uv__run_pending(loop); 

270 # uv__run_idle(loop); 

271 # uv__run_prepare(loop); 

272 # ... 

273 # uv__io_poll(loop, timeout); # <--- IO watchers run here! 

274 # uv__run_check(loop); 

275 

276 # libev looks something like this (pseudo code because the real code is 

277 # hard to read): 

278 # 

279 # do { 

280 # run_fork_callbacks(); 

281 # run_prepare_callbacks(); 

282 # timeout = min(time of all timers or normal block time) 

283 # io_poll() # <--- Only queues IO callbacks 

284 # update_now(); calculate_expired_timers(); 

285 # run callbacks in this order: (although specificying priorities changes it) 

286 # check 

287 # stat 

288 # child 

289 # signal 

290 # timer 

291 # io 

292 # } 

293 

294 # So instead of running a no-op and letting the side-effect of spinning 

295 # the loop run the callbacks, we must explicitly run them here. 

296 

297 # If we don't, test__systemerror:TestCallback will be flaky, failing 

298 # one time out of ~20, depending on timing. 

299 

300 # To get them to run immediately after this current loop, 

301 # we use a check watcher, instead of a 0 duration timer entirely. 

302 # If we use a 0 duration timer, we can get stuck in a timer loop. 

303 # Python 3.6 fails in test_ftplib.py 

304 

305 # As a final note, if we have not yet entered the loop *at 

306 # all*, and a timer was created with a duration shorter than 

307 # the amount of time it took for us to enter the loop in the 

308 # first place, it may expire and get called before our callback 

309 # does. This could also lead to test__systemerror:TestCallback 

310 # appearing to be flaky. 

311 

312 # As yet another final note, if we are currently running a 

313 # timer callback, meaning we're inside uv__run_timers() in C, 

314 # and the Python starts a new timer, if the Python code then 

315 # update's the loop's time, it's possible that timer will 

316 # expire *and be run in the same iteration of the loop*. This 

317 # is trivial to do: In sequential code, anything after 

318 # `gevent.sleep(0.1)` is running in a timer callback. Starting 

319 # a new timer---e.g., another gevent.sleep() call---will 

320 # update the time, *before* uv__run_timers exits, meaning 

321 # other timers get a chance to run before our check or prepare 

322 # watcher callbacks do. Therefore, we do indeed have to have a 0 

323 # timer to run callbacks---it gets inserted before any other user 

324 # timers---ideally, this should be especially careful about how much time 

325 # it runs for. 

326 

327 # AND YET: We can't actually do that. We get timeouts that I haven't fully 

328 # investigated if we do. Probably stuck in a timer loop. 

329 

330 # As a partial remedy to this, unlike libev, our timer watcher 

331 # class doesn't update the loop time by default. 

332 

333 libuv.uv_check_start(self._timer0, libuv.python_timer0_callback)

487 def __run_queued_callbacks(self): 

488 if not self._queued_callbacks: 

489 return False 

490 

491 cbs = self._queued_callbacks[:] 

492 del self._queued_callbacks[:] 

493 

494 for watcher_ptr, arg in cbs: 

495 handle = watcher_ptr.data 

496 if not handle: 

497 # It's been stopped and possibly closed 

498 assert not libuv.uv_is_active(watcher_ptr) 

499 continue 

500 val = _callbacks.python_callback(handle, arg) 

501 if val == -1: # Failure. 

502 _callbacks.python_handle_error(handle, arg) 

503 elif val == 1: # Success, and we may need to close the Python watcher. 

504 if not libuv.uv_is_active(watcher_ptr): 

505 # The callback closed the native watcher resources. Good. 

506 # It's *supposed* to also reset the .data handle to NULL at 

507 # that same time. If it resets it to something else, we're 

508 # re-using the same watcher object, and that's not correct either. 

509 # On Windows in particular, if the .data handle is changed because 

510 # the IO multiplexer is being restarted, trying to dereference the 

511 # *old* handle can crash with an FFI error. 

512 handle_after_callback = watcher_ptr.data 

513 try: 

514 if handle_after_callback and handle_after_callback == handle: 

515 _callbacks.python_stop(handle_after_callback) 

516 finally: 

517 watcher_ptr.data = ffi.NULL 

518 return True

521 def run(self, nowait=False, once=False): 

522 # we can only respect one flag or the other. 

523 # nowait takes precedence because it can't block 

524 mode = libuv.UV_RUN_DEFAULT 

525 if once: 

526 mode = libuv.UV_RUN_ONCE 

527 if nowait: 

528 mode = libuv.UV_RUN_NOWAIT 

529 

530 if mode == libuv.UV_RUN_DEFAULT: 

531 while self._ptr and self._ptr.data: 

532 # This is here to better preserve order guarantees. 

533 # See _run_callbacks for details. 

534 

535 # It may get run again from the prepare watcher, so 

536 # potentially we could take twice as long as the 

537 # switch interval. 

538 # If we have *lots* of callbacks to run, we may not actually 

539 # get through them all before we're requested to poll for IO; 

540 # so in that case, just spin the loop once (UV_RUN_NOWAIT) and 

541 # go again. 

542 self._run_callbacks() 

543 self._prepare_ran_callbacks = False 

544 

545 # UV_RUN_ONCE will poll for IO, blocking for up to the time needed 

546 # for the next timer to expire. Worst case, that's our _signal_idle 

547 # timer, about 1/3 second. UV_RUN_ONCE guarantees that some forward progress 

548 # is made, either by an IO watcher or a timer. 

549 # 

550 # In contrast, UV_RUN_NOWAIT makes no such guarantee, it only polls for IO once and 

551 # immediately returns; it does not update the loop time or timers after 

552 # polling for IO. 

553 run_mode = ( 

554 libuv.UV_RUN_ONCE 

555 if not self._callbacks and not self._queued_callbacks 

556 else libuv.UV_RUN_NOWAIT 

557 ) 

558 

559 ran_status = libuv.uv_run(self._ptr, run_mode) 

560 # Note that we run queued callbacks when the prepare watcher runs, 

561 # thus accounting for timers that expired before polling for IO, 

562 # and idle watchers. This next call should get IO callbacks and 

563 # callbacks from timers that expired *after* polling for IO. 

564 ran_callbacks = self.__run_queued_callbacks() 

565 

566 if not ran_status and not ran_callbacks and not self._prepare_ran_callbacks: 

567 # A return of 0 means there are no referenced and 

568 # active handles. The loop is over. 

569 # If we didn't run any callbacks, then we couldn't schedule 

570 # anything to switch in the future, so there's no point 

571 # running again. 

572 return ran_status 

573 return 0 # Somebody closed the loop 

574 

575 result = libuv.uv_run(self._ptr, mode) 

576 self.__run_queued_callbacks() 

577 return result

617 def reset_sigchld(self): 

618 if not self.default or not self._sigchld_watcher: 

619 return 

620 

621 libuv.uv_signal_stop(self._sigchld_watcher) 

622 # Must go through this to manage the memory lifetime 

623 # correctly. Alternately, we could just stop it and restart 

624 # it in install_sigchld? 

625 _watchers.watcher._watcher_ffi_close(self._sigchld_watcher) 

626 del self._sigchld_watcher 

627 

628 

629 def _sigchld_callback(self): 

630 # Signals can arrive at (relatively) any time. To eliminate 

631 # race conditions, and behave more like libev, we "queue" 

632 # sigchld to run when we run callbacks. 

633 while True: 

634 try: 

635 pid, status, _usage = os.wait3(os.WNOHANG) 

636 except OSError: 

637 # Python 3 raises ChildProcessError 

638 break 

639 

640 if pid == 0: 

641 break 

642 children_watchers = self._child_watchers.get(pid, []) + self._child_watchers.get(0, []) 

643 for watcher in children_watchers: 

644 self.run_callback(watcher._set_waitpid_status, pid, status) 

645 

646 # Don't invoke child watchers for 0 more than once 

647 self._child_watchers[0] = [] 

648 

649 def _register_child_watcher(self, watcher): 

650 self._child_watchers[watcher._pid].append(watcher) 

651 

652 def _unregister_child_watcher(self, watcher): 

653 try: 

654 # stop() should be idempotent 

655 self._child_watchers[watcher._pid].remove(watcher) 

656 except ValueError: 

657 pass 

658 

659 # Now's a good time to clean up any dead watchers we don't need 

660 # anymore 

661 for pid in list(self._child_watchers): 

662 if not self._child_watchers[pid]: 

663 del self._child_watchers[pid] 

664 

665 def io(self, fd, events, ref=True, priority=None): 

666 # We rely on hard references here and explicit calls to 

667 # close() on the returned object to correctly manage 

668 # the watcher lifetimes. 

669 

670 io_watchers = self._io_watchers 

671 try: 

672 io_watcher = io_watchers[fd] 

673 assert io_watcher._multiplex_watchers, ("IO Watcher %s unclosed but should be dead" % io_watcher) 

674 except KeyError: 

675 # Start the watcher with just the events that we're interested in. 

676 # as multiplexers are added, the real event mask will be updated to keep in sync. 

677 # If we watch for too much, we get spurious wakeups and busy loops. 

678 io_watcher = self._watchers.io(self, fd, 0) 

679 io_watchers[fd] = io_watcher 

680 io_watcher._no_more_watchers = lambda: delitem(io_watchers, fd) 

681 

682 return io_watcher.multiplex(events)

684 def prepare(self, ref=True, priority=None): 

685 # We run arbitrary code in python_prepare_callback. That could switch 

686 # greenlets. If it does that while also manipulating the active prepare 

687 # watchers, we could corrupt the process state, since the prepare watcher 

688 # queue is iterated on the stack (on unix). We could workaround this by implementing 

689 # prepare watchers in pure Python. 

690 # See https://github.com/gevent/gevent/issues/1126 

691 raise TypeError("prepare watchers are not currently supported in libuv. " 

692 "If you need them, please contact the maintainers.")
