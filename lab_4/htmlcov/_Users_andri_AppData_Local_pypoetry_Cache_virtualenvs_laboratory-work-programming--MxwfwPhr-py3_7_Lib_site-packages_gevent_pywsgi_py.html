127 def _discard(self): 

128 if self._chunked_input_error: 

129 # We are in an unknown state, so we can't necessarily discard 

130 # the body (e.g., if the client keeps the socket open, we could hang 

131 # here forever). 

132 # In this case, we've raised an exception and the user of this object 

133 # is going to close the socket, so we don't have to discard 

134 return 

135 

136 if self.socket is None and (self.position < (self.content_length or 0) or self.chunked_input): 

137 # ## Read and discard body 

138 while 1: 

139 d = self.read(16384) 

140 if not d: 

141 break

148 def _do_read(self, length=None, use_readline=False): 

149 if use_readline: 

150 reader = self.rfile.readline 

151 else: 

152 reader = self.rfile.read 

153 content_length = self.content_length 

154 if content_length is None: 

155 # Either Content-Length or "Transfer-Encoding: chunked" must be present in a request with a body 

156 # if it was chunked, then this function would have not been called 

157 return b'' 

158 

159 self._send_100_continue() 

160 left = content_length - self.position 

161 if length is None: 

162 length = left 

163 elif length > left: 

164 length = left 

165 if not length: 

166 return b'' 

167 

168 # On Python 2, self.rfile is usually socket.makefile(), which 

169 # uses cStringIO.StringIO. If *length* is greater than the C 

170 # sizeof(int) (typically 32 bits signed), parsing the argument to 

171 # readline raises OverflowError. StringIO.read(), OTOH, uses 

172 # PySize_t, typically a long (64 bits). In a bare readline() 

173 # case, because the header lines we're trying to read with 

174 # readline are typically expected to be small, we can correct 

175 # that failure by simply doing a smaller call to readline and 

176 # appending; failures in read we let propagate. 

177 try: 

178 read = reader(length) 

179 except OverflowError: 

180 if not use_readline: 

181 # Expecting to read more than 64 bits of data. Ouch! 

182 raise 

183 # We could loop on calls to smaller readline(), appending them 

184 # until we actually get a newline. For uses in this module, 

185 # we expect the actual length to be small, but WSGI applications 

186 # are allowed to pass in an arbitrary length. (This loop isn't optimal, 

187 # but even client applications *probably* have short lines.) 

188 read = b'' 

189 while len(read) < length and not read.endswith(b'\n'): 

190 read += reader(MAX_REQUEST_LINE) 

191 

192 self.position += len(read) 

193 if len(read) < length: 

194 if (use_readline and not read.endswith(b"\n")) or not use_readline: 

195 raise IOError("unexpected end of file while reading request at position %s" % (self.position,)) 

196 

197 return read

199 def __read_chunk_length(self, rfile): 

200 # Read and return the next integer chunk length. If no 

201 # chunk length can be read, raises _InvalidClientInput. 

202 

203 # Here's the production for a chunk: 

204 # (http://www.w3.org/Protocols/rfc2616/rfc2616-sec3.html) 

205 # chunk = chunk-size [ chunk-extension ] CRLF 

206 # chunk-data CRLF 

207 # chunk-size = 1*HEX 

208 # chunk-extension= *( ";" chunk-ext-name [ "=" chunk-ext-val ] ) 

209 # chunk-ext-name = token 

210 # chunk-ext-val = token | quoted-string 

211 

212 # To cope with malicious or broken clients that fail to send valid 

213 # chunk lines, the strategy is to read character by character until we either reach 

214 # a ; or newline. If at any time we read a non-HEX digit, we bail. If we hit a 

215 # ;, indicating an chunk-extension, we'll read up to the next 

216 # MAX_REQUEST_LINE characters 

217 # looking for the CRLF, and if we don't find it, we bail. If we read more than 16 hex characters, 

218 # (the number needed to represent a 64-bit chunk size), we bail (this protects us from 

219 # a client that sends an infinite stream of `F`, for example). 

220 

221 buf = BytesIO() 

222 while 1: 

223 char = rfile.read(1) 

224 if not char: 

225 self._chunked_input_error = True 

226 raise _InvalidClientInput("EOF before chunk end reached") 

227 if char == b'\r': 

228 break 

229 if char == b';': 

230 break 

231 

232 if char not in _HEX: 

233 self._chunked_input_error = True 

234 raise _InvalidClientInput("Non-hex data", char) 

235 buf.write(char) 

236 if buf.tell() > 16: 

237 self._chunked_input_error = True 

238 raise _InvalidClientInput("Chunk-size too large.") 

239 

240 if char == b';': 

241 i = 0 

242 while i < MAX_REQUEST_LINE: 

243 char = rfile.read(1) 

244 if char == b'\r': 

245 break 

246 i += 1 

247 else: 

248 # we read more than MAX_REQUEST_LINE without 

249 # hitting CR 

250 self._chunked_input_error = True 

251 raise _InvalidClientInput("Too large chunk extension") 

252 

253 if char == b'\r': 

254 # We either got here from the main loop or from the 

255 # end of an extension 

256 char = rfile.read(1) 

257 if char != b'\n': 

258 self._chunked_input_error = True 

259 raise _InvalidClientInput("Line didn't end in CRLF") 

260 return int(buf.getvalue(), 16)

262 def _chunked_read(self, length=None, use_readline=False): 

263 # pylint:disable=too-many-branches 

264 rfile = self.rfile 

265 self._send_100_continue() 

266 

267 if length == 0: 

268 return b"" 

269 

270 if use_readline: 

271 reader = self.rfile.readline 

272 else: 

273 reader = self.rfile.read 

274 

275 response = [] 

276 while self.chunk_length != 0: 

277 maxreadlen = self.chunk_length - self.position 

278 if length is not None and length < maxreadlen: 

279 maxreadlen = length 

280 

281 if maxreadlen > 0: 

282 data = reader(maxreadlen) 

283 if not data: 

284 self.chunk_length = 0 

285 self._chunked_input_error = True 

286 raise IOError("unexpected end of file while parsing chunked data") 

287 

288 datalen = len(data) 

289 response.append(data) 

290 

291 self.position += datalen 

292 if self.chunk_length == self.position: 

293 rfile.readline() 

294 

295 if length is not None: 

296 length -= datalen 

297 if length == 0: 

298 break 

299 if use_readline and data[-1] == b"\n"[0]: 

300 break 

301 else: 

302 # We're at the beginning of a chunk, so we need to 

303 # determine the next size to read 

304 self.chunk_length = self.__read_chunk_length(rfile) 

305 self.position = 0 

306 if self.chunk_length == 0: 

307 # Last chunk. Terminates with a CRLF. 

308 rfile.readline() 

309 return b''.join(response)

443 def handle(self): 

444 """ 

445 The main request handling method, called by the server. 

446 

447 This method runs a request handling loop, calling 

448 :meth:`handle_one_request` until all requests on the 

449 connection have been handled (that is, it implements 

450 keep-alive). 

451 """ 

452 try: 

453 while self.socket is not None: 

454 self.time_start = time.time() 

455 self.time_finish = 0 

456 

457 result = self.handle_one_request() 

458 if result is None: 

459 break 

460 if result is True: 

461 continue 

462 

463 self.status, response_body = result 

464 self.socket.sendall(response_body) 

465 if self.time_finish == 0: 

466 self.time_finish = time.time() 

467 self.log_request() 

468 break 

469 finally: 

470 if self.socket is not None: 

471 _sock = getattr(self.socket, '_sock', None) # Python 3 

472 try: 

473 # read out request data to prevent error: [Errno 104] Connection reset by peer 

474 if _sock: 

475 try: 

476 # socket.recv would hang 

477 _sock.recv(16384) 

478 finally: 

479 _sock.close() 

480 self.socket.close() 

481 except socket.error: 

482 pass 

483 self.__dict__.pop('socket', None) 

484 self.__dict__.pop('rfile', None)

495 def read_request(self, raw_requestline): 

496 """ 

497 Parse the incoming request. 

498 

499 Parses various headers into ``self.headers`` using 

500 :attr:`MessageClass`. Other attributes that are set upon a successful 

501 return of this method include ``self.content_length`` and ``self.close_connection``. 

502 

503 :param str raw_requestline: A native :class:`str` representing 

504 the request line. A processed version of this will be stored 

505 into ``self.requestline``. 

506 

507 :raises ValueError: If the request is invalid. This error will 

508 not be logged as a traceback (because it's a client issue, not a server problem). 

509 :return: A boolean value indicating whether the request was successfully parsed. 

510 This method should either return a true value or have raised a ValueError 

511 with details about the parsing error. 

512 

513 .. versionchanged:: 1.1b6 

514 Raise the previously documented :exc:`ValueError` in more cases instead of returning a 

515 false value; this allows subclasses more opportunity to customize behaviour. 

516 """ 

517 # pylint:disable=too-many-branches 

518 self.requestline = raw_requestline.rstrip() 

519 words = self.requestline.split() 

520 if len(words) == 3: 

521 self.command, self.path, self.request_version = words 

522 if not self._check_http_version(): 

523 raise _InvalidClientRequest('Invalid http version: %r' % (raw_requestline,)) 

524 elif len(words) == 2: 

525 self.command, self.path = words 

526 if self.command != "GET": 

527 raise _InvalidClientRequest('Expected GET method: %r' % (raw_requestline,)) 

528 self.request_version = "HTTP/0.9" 

529 # QQQ I'm pretty sure we can drop support for HTTP/0.9 

530 else: 

531 raise _InvalidClientRequest('Invalid HTTP method: %r' % (raw_requestline,)) 

532 

533 self.headers = self.MessageClass(self.rfile, 0) 

534 

535 if self.headers.status: 

536 raise _InvalidClientRequest('Invalid headers status: %r' % (self.headers.status,)) 

537 

538 if self.headers.get("transfer-encoding", "").lower() == "chunked": 

539 try: 

540 del self.headers["content-length"] 

541 except KeyError: 

542 pass 

543 

544 content_length = self.headers.get("content-length") 

545 if content_length is not None: 

546 content_length = int(content_length) 

547 if content_length < 0: 

548 raise _InvalidClientRequest('Invalid Content-Length: %r' % (content_length,)) 

549 

550 if content_length and self.command in ('HEAD', ): 

551 raise _InvalidClientRequest('Unexpected Content-Length') 

552 

553 self.content_length = content_length 

554 

555 if self.request_version == "HTTP/1.1": 

556 conntype = self.headers.get("Connection", "").lower() 

557 self.close_connection = (conntype == 'close') 

558 elif self.request_version == 'HTTP/1.0': 

559 conntype = self.headers.get("Connection", "close").lower() 

560 self.close_connection = (conntype != 'keep-alive') 

561 else: 

562 # XXX: HTTP 0.9. We should drop support 

563 self.close_connection = True 

564 

565 return True

597 def handle_one_request(self): 

598 """ 

599 Handles one HTTP request using ``self.socket`` and ``self.rfile``. 

600 

601 Each invocation of this method will do several things, including (but not limited to): 

602 

603 - Read the request line using :meth:`read_requestline`; 

604 - Read the rest of the request, including headers, with :meth:`read_request`; 

605 - Construct a new WSGI environment in ``self.environ`` using :meth:`get_environ`; 

606 - Store the application in ``self.application``, retrieving it from the server; 

607 - Handle the remainder of the request, including invoking the application, 

608 with :meth:`handle_one_response` 

609 

610 There are several possible return values to indicate the state 

611 of the client connection: 

612 

613 - ``None`` 

614 The client connection is already closed or should 

615 be closed because the WSGI application or client set the 

616 ``Connection: close`` header. The request handling 

617 loop should terminate and perform cleanup steps. 

618 - (status, body) 

619 An HTTP status and body tuple. The request was in error, 

620 as detailed by the status and body. The request handling 

621 loop should terminate, close the connection, and perform 

622 cleanup steps. Note that the ``body`` is the complete contents 

623 to send to the client, including all headers and the initial 

624 status line. 

625 - ``True`` 

626 The literal ``True`` value. The request was successfully handled 

627 and the response sent to the client by :meth:`handle_one_response`. 

628 The connection remains open to process more requests and the connection 

629 handling loop should call this method again. This is the typical return 

630 value. 

631 

632 .. seealso:: :meth:`handle` 

633 

634 .. versionchanged:: 1.1b6 

635 Funnel exceptions having to do with invalid HTTP requests through 

636 :meth:`_handle_client_error` to allow subclasses to customize. Note that 

637 this is experimental and may change in the future. 

638 """ 

639 # pylint:disable=too-many-return-statements 

640 if self.rfile.closed: 

641 return 

642 

643 try: 

644 self.requestline = self.read_requestline() 

645 # Account for old subclasses that haven't done this 

646 if PY3 and isinstance(self.requestline, bytes): 

647 self.requestline = self.requestline.decode('latin-1') 

648 except socket.error: 

649 # "Connection reset by peer" or other socket errors aren't interesting here 

650 return 

651 

652 if not self.requestline: 

653 return 

654 

655 self.response_length = 0 

656 

657 if len(self.requestline) >= MAX_REQUEST_LINE: 

658 return ('414', _REQUEST_TOO_LONG_RESPONSE) 

659 

660 try: 

661 # for compatibility with older versions of pywsgi, we pass self.requestline as an argument there 

662 # NOTE: read_request is supposed to raise ValueError on invalid input; allow old 

663 # subclasses that return a False value instead. 

664 # NOTE: This can mutate the value of self.headers, so self.get_environ() must not be 

665 # called until AFTER this call is done. 

666 if not self.read_request(self.requestline): 

667 return ('400', _BAD_REQUEST_RESPONSE) 

668 except Exception as ex: # pylint:disable=broad-except 

669 # Notice we don't use self.handle_error because it reports 

670 # a 500 error to the client, and this is almost certainly 

671 # a client error. 

672 # Provide a hook for subclasses. 

673 return self._handle_client_error(ex) 

674 

675 self.environ = self.get_environ() 

676 self.application = self.server.application 

677 

678 self.handle_one_response() 

679 

680 if self.close_connection: 

681 return 

682 

683 if self.rfile.closed: 

684 return 

685 

686 return True # read more requests

688 def finalize_headers(self): 

689 if self.provided_date is None: 

690 self.response_headers.append((b'Date', format_date_time(time.time()))) 

691 

692 if self.code not in (304, 204): 

693 # the reply will include message-body; make sure we have either Content-Length or chunked 

694 if self.provided_content_length is None: 

695 if hasattr(self.result, '__len__'): 

696 total_len = sum(len(chunk) for chunk in self.result) 

697 total_len_str = str(total_len) 

698 if PY3: 

699 total_len_str = total_len_str.encode("latin-1") 

700 self.response_headers.append((b'Content-Length', total_len_str)) 

701 else: 

702 if self.request_version != 'HTTP/1.0': 

703 self.response_use_chunked = True 

704 self.response_headers.append((b'Transfer-Encoding', b'chunked'))

706 def _sendall(self, data): 

707 try: 

708 self.socket.sendall(data) 

709 except socket.error as ex: 

710 self.status = 'socket error: %s' % ex 

711 if self.code > 0: 

712 self.code = -self.code 

713 raise 

714 self.response_length += len(data)

716 def _write(self, data, 

717 _bytearray=bytearray): 

718 if not data: 

719 # The application/middleware are allowed to yield 

720 # empty bytestrings. 

721 return 

722 

723 if self.response_use_chunked: 

724 # Write the chunked encoding header 

725 header_str = b'%x\r\n' % len(data) 

726 towrite = _bytearray(header_str) 

727 

728 # data 

729 towrite += data 

730 # trailer 

731 towrite += b'\r\n' 

732 self._sendall(towrite) 

733 else: 

734 self._sendall(data)

738 def write(self, data): 

739 # The write() callable we return from start_response. 

740 # https://www.python.org/dev/peps/pep-3333/#the-write-callable 

741 # Supposed to do pretty much the same thing as yielding values 

742 # from the application's return. 

743 if self.code in (304, 204) and data: 

744 raise self.ApplicationError('The %s response must have no body' % self.code) 

745 

746 if self.headers_sent: 

747 self._write(data) 

748 else: 

749 if not self.status: 

750 raise self.ApplicationError("The application did not call start_response()") 

751 self._write_with_headers(data)

753 def _write_with_headers(self, data): 

754 self.headers_sent = True 

755 self.finalize_headers() 

756 

757 # self.response_headers and self.status are already in latin-1, as encoded by self.start_response 

758 towrite = bytearray(b'HTTP/1.1 ') 

759 towrite += self.status 

760 towrite += b'\r\n' 

761 for header, value in self.response_headers: 

762 towrite += header 

763 towrite += b': ' 

764 towrite += value 

765 towrite += b"\r\n" 

766 

767 towrite += b'\r\n' 

768 self._sendall(towrite) 

769 # No need to copy the data into towrite; we may make an extra syscall 

770 # but the copy time could be substantial too, and it reduces the chances 

771 # of sendall being able to send everything in one go 

772 self._write(data)

774 def start_response(self, status, headers, exc_info=None): 

775 """ 

776 .. versionchanged:: 1.2a1 

777 Avoid HTTP header injection by raising a :exc:`ValueError` 

778 if *status* or any *header* name or value contains a carriage 

779 return or newline. 

780 .. versionchanged:: 1.1b5 

781 Pro-actively handle checking the encoding of the status line 

782 and headers during this method. On Python 2, avoid some 

783 extra encodings. 

784 """ 

785 # pylint:disable=too-many-branches,too-many-statements 

786 if exc_info: 

787 try: 

788 if self.headers_sent: 

789 # Re-raise original exception if headers sent 

790 reraise(*exc_info) 

791 finally: 

792 # Avoid dangling circular ref 

793 exc_info = None 

794 

795 # Pep 3333, "The start_response callable": 

796 # https://www.python.org/dev/peps/pep-3333/#the-start-response-callable 

797 # "Servers should check for errors in the headers at the time 

798 # start_response is called, so that an error can be raised 

799 # while the application is still running." Here, we check the encoding. 

800 # This aids debugging: headers especially are generated programmatically 

801 # and an encoding error in a loop or list comprehension yields an opaque 

802 # UnicodeError without any clue which header was wrong. 

803 # Note that this results in copying the header list at this point, not modifying it, 

804 # although we are allowed to do so if needed. This slightly increases memory usage. 

805 # We also check for HTTP Response Splitting vulnerabilities 

806 response_headers = [] 

807 header = None 

808 value = None 

809 try: 

810 for header, value in headers: 

811 if not isinstance(header, str): 

812 raise UnicodeError("The header must be a native string", header, value) 

813 if not isinstance(value, str): 

814 raise UnicodeError("The value must be a native string", header, value) 

815 if '\r' in header or '\n' in header: 

816 raise ValueError('carriage return or newline in header name', header) 

817 if '\r' in value or '\n' in value: 

818 raise ValueError('carriage return or newline in header value', value) 

819 # Either we're on Python 2, in which case bytes is correct, or 

820 # we're on Python 3 and the user screwed up (because it should be a native 

821 # string). In either case, make sure that this is latin-1 compatible. Under 

822 # Python 2, bytes.encode() will take a round-trip through the system encoding, 

823 # which may be ascii, which is not really what we want. However, the latin-1 encoding 

824 # can encode everything except control characters and the block from 0x7F to 0x9F, so 

825 # explicitly round-tripping bytes through the encoding is unlikely to be of much 

826 # benefit, so we go for speed (the WSGI spec specifically calls out allowing the range 

827 # from 0x00 to 0xFF, although the HTTP spec forbids the control characters). 

828 # Note: Some Python 2 implementations, like Jython, may allow non-octet (above 255) values 

829 # in their str implementation; this is mentioned in the WSGI spec, but we don't 

830 # run on any platform like that so we can assume that a str value is pure bytes. 

831 response_headers.append((header if not PY3 else header.encode("latin-1"), 

832 value if not PY3 else value.encode("latin-1"))) 

833 except UnicodeEncodeError: 

834 # If we get here, we're guaranteed to have a header and value 

835 raise UnicodeError("Non-latin1 header", repr(header), repr(value)) 

836 

837 # Same as above 

838 if not isinstance(status, str): 

839 raise UnicodeError("The status string must be a native string") 

840 if '\r' in status or '\n' in status: 

841 raise ValueError("carriage return or newline in status", status) 

842 # don't assign to anything until the validation is complete, including parsing the 

843 # code 

844 code = int(status.split(' ', 1)[0]) 

845 

846 self.status = status if not PY3 else status.encode("latin-1") 

847 self._orig_status = status # Preserve the native string for logging 

848 self.response_headers = response_headers 

849 self.code = code 

850 

851 provided_connection = None # Did the wsgi app give us a Connection header? 

852 self.provided_date = None 

853 self.provided_content_length = None 

854 

855 for header, value in headers: 

856 header = header.lower() 

857 if header == 'connection': 

858 provided_connection = value 

859 elif header == 'date': 

860 self.provided_date = value 

861 elif header == 'content-length': 

862 self.provided_content_length = value 

863 

864 if self.request_version == 'HTTP/1.0' and provided_connection is None: 

865 conntype = b'close' if self.close_connection else b'keep-alive' 

866 response_headers.append((b'Connection', conntype)) 

867 elif provided_connection == 'close': 

868 self.close_connection = True 

869 

870 if self.code in (304, 204): 

871 if self.provided_content_length is not None and self.provided_content_length != '0': 

872 msg = 'Invalid Content-Length for %s response: %r (must be absent or zero)' % (self.code, self.provided_content_length) 

873 if PY3: 

874 msg = msg.encode('latin-1') 

875 raise self.ApplicationError(msg) 

876 

877 return self.write

882 def format_request(self): 

883 now = datetime.now().replace(microsecond=0) 

884 length = self.response_length or '-' 

885 if self.time_finish: 

886 delta = '%.6f' % (self.time_finish - self.time_start) 

887 else: 

888 delta = '-' 

889 client_address = self.client_address[0] if isinstance(self.client_address, tuple) else self.client_address

891 client_address or '-',

893 self.requestline or '',

897 (self._orig_status or self.status or '000').split()[0],

899 delta)

927 # all the way through it---the frames are automatically 

928 # cleaned up when StopIteration is raised); but other cases

942 #: This is a provisional API, subject to change. See :pr:`377`, :pr:`999` 

943 #: and :issue:`136`. 

944 #:

953 """ 

954 Invoke the application to produce one response. 

955 

956 This is called by :meth:`handle_one_request` after all the 

957 state for the request has been established. It is responsible 

958 for error handling. 

959 """ 

960 self.time_start = time.time() 

961 self.status = None 

962 self.headers_sent = False 

963 

964 self.result = None 

965 self.response_use_chunked = False 

966 self.response_length = 0

969 try: 

970 self.run_application() 

971 finally: 

972 try: 

973 self.wsgi_input._discard() 

974 except (socket.error, IOError):

976 # input override any exception that may have been 

977 # raised by the application, such as our own _InvalidClientInput.

987 sys.exc_clear()

991 except: # pylint:disable=bare-except 

992 self.handle_error(*sys.exc_info())

994 self.time_finish = time.time()

1040 def _headers(self): 

1041 key = None 

1042 value = None 

1043 IGNORED_KEYS = (None, 'CONTENT_TYPE', 'CONTENT_LENGTH') 

1044 for header in self.headers.headers: 

1045 if key is not None and header[:1] in " \t": 

1046 value += header 

1047 continue 

1048 

1049 if key not in IGNORED_KEYS: 

1050 yield 'HTTP_' + key, value.strip() 

1051 

1052 key, value = header.split(':', 1) 

1053 if '_' in key: 

1054 # strip incoming bad veaders 

1055 key = None 

1056 else: 

1057 key = key.replace('-', '_').upper() 

1058 

1059 if key not in IGNORED_KEYS: 

1060 yield 'HTTP_' + key, value.strip()

1062 def get_environ(self): 

1063 """ 

1064 Construct and return a new WSGI environment dictionary for a specific request. 

1065 

1066 This should begin with asking the server for the base environment 

1067 using :meth:`WSGIServer.get_environ`, and then proceed to add the 

1068 request specific values. 

1069 

1070 By the time this method is invoked the request line and request shall have 

1071 been parsed and ``self.headers`` shall be populated. 

1072 """ 

1073 env = self.server.get_environ() 

1074 env['REQUEST_METHOD'] = self.command 

1075 env['SCRIPT_NAME'] = '' 

1076 

1077 if '?' in self.path: 

1078 path, query = self.path.split('?', 1) 

1079 else: 

1080 path, query = self.path, '' 

1081 # Note that self.path contains the original str object; if it contains 

1082 # encoded escapes, it will NOT match PATH_INFO. 

1083 env['PATH_INFO'] = unquote_latin1(path) 

1084 env['QUERY_STRING'] = query 

1085 

1086 if self.headers.typeheader is not None: 

1087 env['CONTENT_TYPE'] = self.headers.typeheader 

1088 

1089 length = self.headers.getheader('content-length') 

1090 if length: 

1091 env['CONTENT_LENGTH'] = length 

1092 env['SERVER_PROTOCOL'] = self.request_version 

1093 

1094 client_address = self.client_address 

1095 if isinstance(client_address, tuple): 

1096 env['REMOTE_ADDR'] = str(client_address[0]) 

1097 env['REMOTE_PORT'] = str(client_address[1]) 

1098 

1099 for key, value in self._headers(): 

1100 if key in env: 

1101 if 'COOKIE' in key: 

1102 env[key] += '; ' + value 

1103 else: 

1104 env[key] += ',' + value 

1105 else: 

1106 env[key] = value 

1107 

1108 if env.get('HTTP_EXPECT') == '100-continue': 

1109 sock = self.socket 

1110 else: 

1111 sock = None 

1112 

1113 chunked = env.get('HTTP_TRANSFER_ENCODING', '').lower() == 'chunked' 

1114 self.wsgi_input = Input(self.rfile, self.content_length, socket=sock, chunked_input=chunked) 

1115 env['wsgi.input'] = self.wsgi_input 

1116 # This is a non-standard flag indicating that our input stream is 

1117 # self-terminated (returns EOF when consumed). 

1118 # See https://github.com/gevent/gevent/issues/1308 

1119 env['wsgi.input_terminated'] = True 

1120 return env

1203## Environ classes. 

1204# These subclass dict. They could subclass collections.UserDict on 

1205# 3.3+ and proxy to the underlying real dict to avoid a copy if we 

1206# have to print them (on 2.7 it's slightly more complicated to be an 

1207# instance of collections.MutableMapping; UserDict.UserDict isn't.) 

1208# Then we could have either the WSGIHandler.get_environ or the 

1209# WSGIServer.get_environ return one of these proxies, and 

1210# WSGIHandler.run_application would know to access the `environ.data` 

1211# attribute to be able to pass the *real* dict to the application 

1212# (because PEP3333 requires no subclasses, only actual dict objects; 

1213# wsgiref.validator and webob.Request both enforce this). This has the 

1214# advantage of not being fragile if anybody else tries to print/log 

1215# self.environ (and not requiring a copy). However, if there are any 

1216# subclasses of Handler or Server, this could break if they don't know 

1217# to return this type.

1242class SecureEnviron(Environ): 

1243 """ 

1244 An environment that does not print its keys and values 

1245 by default. 

1246 

1247 Provisional API. 

1248 

1249 This is intended to keep potentially sensitive information like 

1250 HTTP authorization and cookies from being inadvertently printed 

1251 or logged. 

1252 

1253 For debugging, each instance can have its *secure_repr* attribute 

1254 set to ``False``, which will cause it to print like a normal dict. 

1255 

1256 When *secure_repr* is ``True`` (the default), then the value of 

1257 the *whitelist_keys* attribute is consulted; if this value is 

1258 true-ish, it should be a container (something that responds to 

1259 ``in``) of key names (typically a list or set). Keys and values in 

1260 this dictionary that are in *whitelist_keys* will then be printed, 

1261 while all other values will be masked. These values may be 

1262 customized on the class by setting the *default_secure_repr* and 

1263 *default_whitelist_keys*, respectively:: 

1264 

1265 >>> environ = SecureEnviron(key='value') 

1266 >>> environ # doctest: +ELLIPSIS 

1267 <pywsgi.SecureEnviron dict (keys: 1) at ... 

1268 

1269 If we whitelist the key, it gets printed:: 

1270 

1271 >>> environ.whitelist_keys = {'key'} 

1272 >>> environ 

1273 {'key': 'value'} 

1274 

1275 A non-whitelisted key (*only*, to avoid doctest issues) is masked:: 

1276 

1277 >>> environ['secure'] = 'secret'; del environ['key'] 

1278 >>> environ 

1279 {'secure': '<MASKED>'} 

1280 

1281 We can turn it off entirely for the instance:: 

1282 

1283 >>> environ.secure_repr = False 

1284 >>> environ 

1285 {'secure': 'secret'} 

1286 

1287 We can also customize it at the class level (here we use a new 

1288 class to be explicit and to avoid polluting the true default 

1289 values; we would set this class to be the ``environ_class`` of the 

1290 server):: 

1291 

1292 >>> class MyEnviron(SecureEnviron): 

1293 ... default_whitelist_keys = ('key',) 

1294 ... 

1295 >>> environ = MyEnviron({'key': 'value'}) 

1296 >>> environ 

1297 {'key': 'value'} 

1298 

1299 .. versionadded:: 1.2a1 

1300 """ 

1301 

1302 default_secure_repr = True 

1303 default_whitelist_keys = () 

1304 default_print_masked_keys = True 

1305 

1306 # Allow instances to override the class values, 

1307 # but inherit from the class if not present. Keeps instances 

1308 # small since we can't combine __slots__ with class attributes 

1309 # of the same name. 

1310 __slots__ = ('secure_repr', 'whitelist_keys', 'print_masked_keys') 

1311 

1312 def __getattr__(self, name): 

1313 if name in SecureEnviron.__slots__: 

1314 return getattr(type(self), 'default_' + name) 

1315 raise AttributeError(name) 

1316 

1317 def __repr__(self): 

1318 if self.secure_repr: 

1319 whitelist = self.whitelist_keys 

1320 print_masked = self.print_masked_keys 

1321 if whitelist: 

1322 safe = {k: self[k] if k in whitelist else "<MASKED>" 

1323 for k in self 

1324 if k in whitelist or print_masked} 

1325 safe_repr = repr(safe) 

1326 if not print_masked and len(safe) != len(self): 

1327 safe_repr = safe_repr[:-1] + ", (hidden keys: %d)}" % (len(self) - len(safe)) 

1328 return safe_repr 

1329 return "<pywsgi.SecureEnviron dict (keys: %d) at %s>" % (len(self), id(self)) 

1330 return Environ.__repr__(self) 

1331 __str__ = __repr__

1354class WSGIServer(StreamServer): 

1355 """ 

1356 A WSGI server based on :class:`StreamServer` that supports HTTPS. 

1357 

1358 

1359 :keyword log: If given, an object with a ``write`` method to which 

1360 request (access) logs will be written. If not given, defaults 

1361 to :obj:`sys.stderr`. You may pass ``None`` to disable request 

1362 logging. You may use a wrapper, around e.g., :mod:`logging`, 

1363 to support objects that don't implement a ``write`` method. 

1364 (If you pass a :class:`~logging.Logger` instance, or in 

1365 general something that provides a ``log`` method but not a 

1366 ``write`` method, such a wrapper will automatically be created 

1367 and it will be logged to at the :data:`~logging.INFO` level.) 

1368 

1369 :keyword error_log: If given, a file-like object with ``write``, 

1370 ``writelines`` and ``flush`` methods to which error logs will 

1371 be written. If not given, defaults to :obj:`sys.stderr`. You 

1372 may pass ``None`` to disable error logging (not recommended). 

1373 You may use a wrapper, around e.g., :mod:`logging`, to support 

1374 objects that don't implement the proper methods. This 

1375 parameter will become the value for ``wsgi.errors`` in the 

1376 WSGI environment (if not already set). (As with *log*, 

1377 wrappers for :class:`~logging.Logger` instances and the like 

1378 will be created automatically and logged to at the :data:`~logging.ERROR` 

1379 level.) 

1380 

1381 .. seealso:: 

1382 

1383 :class:`LoggingLogAdapter` 

1384 See important warnings before attempting to use :mod:`logging`. 

1385 

1386 .. versionchanged:: 1.1a3 

1387 Added the ``error_log`` parameter, and set ``wsgi.errors`` in the WSGI 

1388 environment to this value. 

1389 .. versionchanged:: 1.1a3 

1390 Add support for passing :class:`logging.Logger` objects to the ``log`` and 

1391 ``error_log`` arguments. 

1392 .. versionchanged:: 20.6.0 

1393 Passing a ``handle`` kwarg to the constructor is now officially deprecated.

1396 #: A callable taking three arguments: (socket, address, server) and returning 

1397 #: an object with a ``handle()`` method. The callable is called once for 

1398 #: each incoming socket request, as is its handle method. The handle method should not 

1399 #: return until all use of the socket is complete. 

1400 #: 

1401 #: This class uses the :class:`WSGIHandler` object as the default value. You may 

1402 #: subclass this class and set a different default value, or you may pass 

1403 #: a value to use in the ``handler_class`` keyword constructor argument.

1408 #: parameter.

1420 #: (probably to :class:`WSGISecureEnviron`).

1426 # will cast to before passing to the loop.

1430 'SERVER_SOFTWARE': 'gevent/%d.%d Python/%d.%d' % (gevent.version_info[:2] + sys.version_info[:2]), 

1431 'SCRIPT_NAME': '', 

1432 'wsgi.version': (1, 0), 

1433 'wsgi.multithread': False, # XXX: Aren't we really, though? 

1434 'wsgi.multiprocess': False, 

1435 'wsgi.run_once': False}

1438 log='default', error_log='default', 

1439 handler_class=None, 

1440 environ=None, **ssl_args): 

1441 if 'handle' in ssl_args:

1505 Called before the first request is handled to fill in WSGI environment values. 

1506
